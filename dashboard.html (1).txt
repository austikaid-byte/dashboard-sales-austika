<script>
// 🔥 PASTE CONFIG FIREBASE DI SINI
const firebaseConfig = {
  apiKey: "ISI_PUNYAMU",
  authDomain: "ISI",
  projectId: "ISI"
};

firebase.initializeApp(firebaseConfig);
const db = firebase.firestore();

let data = [];

async function tambahData(){
    let produk = document.getElementById("produk").value;
    let qty = parseInt(document.getElementById("qty").value);
    let harga = parseInt(document.getElementById("harga").value);

    await db.collection("sales").add({
        produk, qty, harga, created: new Date()
    });
}

db.collection("sales").onSnapshot(snapshot => {
    data = [];
    snapshot.forEach(doc => data.push(doc.data()));
    updateDashboard();
});

function updateDashboard(){
    let total = 0;
    let map = {};

    data.forEach(d => {
        total += d.qty * d.harga;

        if(!map[d.produk]) map[d.produk]=0;
        map[d.produk]+=d.qty;
    });

    document.getElementById("sales").innerText = total;

    renderChart(map);
}

let chart;
function renderChart(map){
    let labels = Object.keys(map);
    let values = Object.values(map);

    if(chart) chart.destroy();

    chart = new Chart(document.getElementById("chart"), {
        type:"bar",
        data:{ labels, datasets:[{ data:values, backgroundColor:"#D4AF37"}]}
    });
}
